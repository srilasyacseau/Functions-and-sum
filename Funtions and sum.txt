#include <stdio.h>
int sum(int a,int b)
{
    int sum;
    sum = a + b;
    return sum;
}

int main()
{
    int x,y,z;
    
    printf("Enter the value of 'x': ");
    scanf("%d",&x);
    printf("Enter the value of 'y': ");
    scanf("%d",&y);
    
    z = sum(x,y);
    
    printf("%d + %d = %d",x,y,z);
    
    return 0;
}
